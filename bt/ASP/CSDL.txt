-- Tạo bảng LoaiDoAn (Loại đồ án)
CREATE TABLE LoaiDoAn (
    MaLoai INT PRIMARY KEY,
    TenLoai NVARCHAR(255)
)

-- Tạo bảng DoAn (Đồ án)
CREATE TABLE DoAn (
    MaDA INT PRIMARY KEY,
    TenDA NVARCHAR(255),
    Mota NVARCHAR(MAX),
    LoaiDA INT,
    FOREIGN KEY (LoaiDA) REFERENCES LoaiDoAn(MaLoai)
)

-- Tạo bảng GiangVien (Giảng viên)
CREATE TABLE GiangVien (
    MaGV INT PRIMARY KEY,
    TenGV NVARCHAR(255),
    TKGV NVARCHAR(50),
    MKGV NVARCHAR(100),
)

-- Tạo bảng SinhVien (Sinh viên)
CREATE TABLE SinhVien (
    MaSV INT PRIMARY KEY,
    TenSV NVARCHAR(255),
    TKSV NVARCHAR(50),
    MKSV NVARCHAR(100),
)

-- Tạo bảng SinhVienDoAn (Sinh viên tham gia đồ án)
CREATE TABLE SinhVienDoAn (
    ID INT PRIMARY KEY,
    StudentID INT,
    ProjectID INT,
    Status NVARCHAR(50), -- Trạng thái tiến độ
    FOREIGN KEY (StudentID) REFERENCES SinhVien(ID),
    FOREIGN KEY (ProjectID) REFERENCES DoAn(ID)
)

-- Tạo bảng GiangVienDoAn (Giảng viên quản lý đồ án)
CREATE TABLE GiangVienDoAn (
    ID INT PRIMARY KEY,
    ProfessorID INT,
    ProjectID INT,
    FOREIGN KEY (ProfessorID) REFERENCES GiangVien(ID),
    FOREIGN KEY (ProjectID) REFERENCES DoAn(ID)
)

-- Tạo bảng Admin (Admin quản trị hệ thống)
CREATE TABLE Admin (
    MaAdmin INT PRIMARY KEY,
    TenAdmin NVARCHAR(255),
    TKadmin NVARCHAR(50),
    MKAdmin NVARCHAR(100),
)

-- Tạo bảng QuyenAdmin (Quyền của admin)
CREATE TABLE QuyenAdmin (
    ID INT PRIMARY KEY,
    AdminID INT,
    Permission NVARCHAR(255),
    FOREIGN KEY (AdminID) REFERENCES Admin(ID)
)
